**2.3.3.4.** 다이얼로그 구현  

　1: getOrderDialogConfig(type) 함수  
　　A. 기본 발주 vs 초과 발주 메시지 분기  

```javascript
const getOrderDialogConfig = (type, capacity) => {
  if (type === 'basic') {
    return {
      title: '발주 신청 확인',
      message: '선택하신 조건으로 컵홀더 발주를 신청합니다.',
      confirmText: '신청',
      cancelText: '취소',
      style: 'confirm'
    };
  }
  const percent = type === '150' ? 150 : 200;
  const quantity = Math.floor(capacity * percent / 100);
  return {
    title: '수용량 초과 발주 안내',
    message: `설정하신 매장 수용량(${capacity}개)을 초과하는 발주입니다.\n선택하신 수량: ${quantity}개 (${percent}%)\n보관 공간이 충분한지 확인해 주세요.`,
    confirmText: '발주 신청',
    cancelText: '취소',
    style: 'warning'
  };
};
```
