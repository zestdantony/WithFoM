```javascript
const validateStep = (step) => {
  switch (step) {
    case 1: // 컵홀더 유형
      return detailInfo.cupholderType.types.length > 0 &&
             detailInfo.cupholderType.holderKind !== '';
    case 2: // 서비스 타입 — 미선택 허용 (선택 사항)
      return true; // 미선택으로도 다음 단계 진행 가능
    case 3: // 상권
      return detailInfo.commercialArea.length >= 2 &&
             detailInfo.commercialArea.length <= 4;
    case 4: // 고객층
      return detailInfo.customerSegments.length >= 2 &&
             detailInfo.customerSegments.length <= 5;
    case 5: // 고객 구성 — 모든 연령대 슬라이더 조작 완료 (초기값에서 변경 여부)
      return Object.values(detailInfo.customerComposition).every(
        ag => ag.male + ag.female === 100
      );
    case 6: // 음료 3종
      return detailInfo.beverages.every(b =>
        b.name.trim() !== '' &&
        b.price >= 1000 &&
        b.price <= 50000
      );
    case 7: // 일일 소진량
      return detailInfo.dailyConsumption >= 10 &&
             detailInfo.dailyConsumption <= 1000 &&
             detailInfo.dailyConsumption % 10 === 0;
    case 8: // 수용량
      const cap = detailInfo.storageCapacity;
      if (detailInfo.cupholderType.holderKind === 'both') {
        return cap.cupholder >= 1000 && cap.cupholder <= 10000 &&
               cap.airholder >= 1000 && cap.airholder <= 10000;
      }
      return cap.cupholder >= 1000 && cap.cupholder <= 10000;
    default:
      return false;
  }
};
```

```javascript
          </View>
          <Slider
            minimumValue={0}
            maximumValue={100}
            step={5}
            value={composition[ag].male}
            onValueChange={(val) => handleSliderChange(ag, val)}
          />
        </View>
      ))}
      <View style={styles.totalRow}>
        <Text>총 남성 비율: {totalMale}%</Text>
```

```javascript
const handleNext = async () => {
  if (!validateStep(currentStep)) {
    showValidationError(currentStep);
    return;
  }

  // 2단계: 서비스 타입 미선택 시 안내 문구 표시
  if (currentStep === 2 && detailInfo.serviceTypes.length === 0) {
    showInfoToast(
      '리워드 기능은 리워드 설정 페이지에서,\n오더 기능은 주문 관리 페이지에서 나중에 활성화할 수 있습니다.'
    );
  }

  // 중간 저장 (선택적)
  await saveProgress(currentStep, detailInfo);

  if (currentStep < 8) {
    setCurrentStep(currentStep + 1);
    setStepCompleted(prev => {
      const newCompleted = [...prev];
      newCompleted[currentStep - 1] = true;
      return newCompleted;
    });
  } else {
    // 마지막 단계 완료
    handleComplete();
  }
};

const handlePrevious = () => {
  if (currentStep > 1) {
    setCurrentStep(currentStep - 1);
  } else {
    // 1단계에서 뒤로가기 — 확인 팝업
    showConfirmDialog(
      '입력한 정보가 저장되지 않습니다. 뒤로 가시겠습니까?',
      () => navigation.goBack()
    );
  }
};
```
